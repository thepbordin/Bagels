    ```python
    @work(thread=True, exclusive=True)
    def run_startup_import(self) -> None:
        """Background worker: optional git pull then full YAML import."""
        try:
            from bagels.config import CONFIG
            if CONFIG is None:
                return

            # Optional: pull from remote before import
            if CONFIG.git.enabled and CONFIG.git.auto_pull:
                try:
                    from bagels.git.operations import pull_from_remote
                    pull_from_remote(
                        remote_name=CONFIG.git.remote,
                        branch=CONFIG.git.branch,
                        silent=True,  # offline/no-remote failures are silent
                    )
                except Exception:
                    pass  # never block startup import due to pull failure

            # Full YAML → SQLite sync
            from bagels.importer.importer import <correct_function_name>
            <correct_function_name>()

            # Notify user on success (non-blocking toast)
            self.app.call_from_thread(
                lambda: self.notify("Data synced from YAML", timeout=2)
            )
        except Exception:
            pass  # Never crash TUI on startup import failure
    ```

    Replace `<correct_function_name>` with the actual import entry point found in Step 1.
